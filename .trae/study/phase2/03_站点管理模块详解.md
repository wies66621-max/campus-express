| 字段名 | 类型 | 说明 | 备注 |
|--------|------|------|------|
| id | BIGINT | 站点ID | 主键，自增 |
| name | VARCHAR(100) | 站点名称 | |
| address | VARCHAR(200) | 站点地址 | |
| phone | VARCHAR(20) | 联系电话 | |
| managerId | BIGINT | 站点管理员ID | 外键 |
| capacity | INT | 容量（件） | |
| status | INT | 状态 | 0:停用, 1:启用 |
| createTime | DATETIME | 创建时间 | |
| updateTime | DATETIME | 更新时间 | |

**SQL建表语句**：
```sql
CREATE TABLE `station` (
  `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '站点ID',
  `name` VARCHAR(100) NOT NULL COMMENT '站点名称',
  `address` VARCHAR(200) COMMENT '站点地址',
  `phone` VARCHAR(20) COMMENT '联系电话',
  `manager_id` BIGINT COMMENT '站点管理员ID',
  `capacity` INT DEFAULT 1000 COMMENT '容量（件）',
  `status` INT NOT NULL DEFAULT 1 COMMENT '状态：0-停用，1-启用',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
  PRIMARY KEY (`id`),
  KEY `idx_manager_id` (`manager_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='站点表';
```

### 2. 站点控制器（StationController.java）

**文件位置**：`src/main/java/com/school/campusexpress/controller/StationController.java`

```java
package com.school.campusexpress.controller;

import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.school.campusexpress.common.Result;
import com.school.campusexpress.entity.Station;
import com.school.campusexpress.service.StationService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/api/station")
@CrossOrigin
public class StationController {
    
    @Autowired
    private StationService stationService;
    
    /**
     * 创建站点
     */
    @PostMapping("/create")
    public Result<?> createStation(@RequestBody Station station) {
        try {
            stationService.createStation(station);
            return Result.success("创建成功");
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 获取站点列表
     */
    @GetMapping("/list")
    public Result<?> getStationList(
        @RequestParam(defaultValue = "1") Integer page,
        @RequestParam(defaultValue = "10") Integer size,
        @RequestParam(required = false) String name,
        @RequestParam(required = false) Integer status
    ) {
        try {
            Page<Station> result = stationService.getStationList(page, size, name, status);
            return Result.success(result);
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 获取所有启用的站点（下拉选择用）
     */
    @GetMapping("/all")
    public Result<?> getAllStations() {
        try {
            return Result.success(stationService.getAllActiveStations());
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 获取站点详情
     */
    @GetMapping("/detail/{id}")
    public Result<?> getStationDetail(@PathVariable Long id) {
        try {
            Station station = stationService.getById(id);
            return Result.success(station);
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 更新站点信息
     */
    @PutMapping("/update")
    public Result<?> updateStation(@RequestBody Station station) {
        try {
            stationService.updateStation(station);
            return Result.success("更新成功");
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 删除站点
     */
    @DeleteMapping("/delete/{id}")
    public Result<?> deleteStation(@PathVariable Long id) {
        try {
            stationService.deleteStation(id);
            return Result.success("删除成功");
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 获取站点统计信息
     */
    @GetMapping("/statistics/{id}")
    public Result<?> getStationStatistics(@PathVariable Long id) {
        try {
            return Result.success(stationService.getStationStatistics(id));
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
}
```

**关键接口说明**：
- `POST /api/station/create`：创建站点
- `GET /api/station/list`：获取站点列表（支持分页和搜索）
- `GET /api/station/all`：获取所有启用的站点（用于下拉选择）
- `GET /api/station/detail/{id}`：获取站点详情
- `PUT /api/station/update`：更新站点信息
- `DELETE /api/station/delete/{id}`：删除站点

### 3. 站点服务实现（StationServiceImpl.java）

**文件位置**：`src/main/java/com/school/campusexpress/service/impl/StationServiceImpl.java`

```java
package com.school.campusexpress.service.impl;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import com.school.campusexpress.entity.Express;
import com.school.campusexpress.entity.Station;
import com.school.campusexpress.mapper.StationMapper;
import com.school.campusexpress.service.ExpressService;
import com.school.campusexpress.service.StationService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.util.StringUtils;

import java.util.HashMap;
import java.util.List;
import java.util.Map;

@Service
public class StationServiceImpl extends ServiceImpl<StationMapper, Station> implements StationService {
    
    @Autowired
    private ExpressService expressService;
    
    @Override
    public void createStation(Station station) {
        // 检查站点名称是否已存在
        LambdaQueryWrapper<Station> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(Station::getName, station.getName());
        Station existStation = getOne(wrapper);
        
        if (existStation != null) {
            throw new RuntimeException("站点名称已存在");
        }
        
        // 设置默认值
        if (station.getCapacity() == null) {
            station.setCapacity(1000);
        }
        
        if (station.getStatus() == null) {
            station.setStatus(1); // 默认启用
        }
        
        save(station);
    }
    
    @Override
    public Page<Station> getStationList(Integer page, Integer size, String name, Integer status) {
        Page<Station> pageParam = new Page<>(page, size);
        LambdaQueryWrapper<Station> wrapper = new LambdaQueryWrapper<>();
        
        if (StringUtils.hasText(name)) {
            wrapper.like(Station::getName, name);
        }
        
        if (status != null) {
            wrapper.eq(Station::getStatus, status);
        }
        
        wrapper.orderByDesc(Station::getCreateTime);
        
        return page(pageParam, wrapper);
    }
    
    @Override
    public List<Station> getAllActiveStations() {
        LambdaQueryWrapper<Station> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(Station::getStatus, 1);
        wrapper.orderByAsc(Station::getName);
        return list(wrapper);
    }
    
    @Override
    public void updateStation(Station station) {
        // 检查站点是否存在
        Station existStation = getById(station.getId());
        if (existStation == null) {
            throw new RuntimeException("站点不存在");
        }
        
        // 检查站点名称是否与其他站点重复
        LambdaQueryWrapper<Station> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(Station::getName, station.getName());
        wrapper.ne(Station::getId, station.getId());
        Station duplicateStation = getOne(wrapper);
        
        if (duplicateStation != null) {
            throw new RuntimeException("站点名称已存在");
        }
        
        updateById(station);
    }
    
    @Override
    public void deleteStation(Long id) {
        // 检查站点是否存在
        Station station = getById(id);
        if (station == null) {
            throw new RuntimeException("站点不存在");
        }
        
        // 检查站点下是否有未取件的快递
        LambdaQueryWrapper<Express> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(Express::getStationId, id);
        wrapper.eq(Express::getStatus, 0); // 待取件
        long count = expressService.count(wrapper);
        
        if (count > 0) {
            throw new RuntimeException("该站点下还有" + count + "件待取件的快递，无法删除");
        }
        
        removeById(id);
    }
    
    @Override
    public Map<String, Object> getStationStatistics(Long id) {
        Map<String, Object> statistics = new HashMap<>();
        
        Station station = getById(id);
        if (station == null) {
            throw new RuntimeException("站点不存在");
        }
        
        // 查询待取件数量
        LambdaQueryWrapper<Express> wrapper1 = new LambdaQueryWrapper<>();
        wrapper1.eq(Express::getStationId, id);
        wrapper1.eq(Express::getStatus, 0);
        long pendingCount = expressService.count(wrapper1);
        
        // 查询已取件数量
        LambdaQueryWrapper<Express> wrapper2 = new LambdaQueryWrapper<>();
        wrapper2.eq(Express::getStationId, id);
        wrapper2.eq(Express::getStatus, 1);
        long pickedCount = expressService.count(wrapper2);
        
        // 查询已取消数量
        LambdaQueryWrapper<Express> wrapper3 = new LambdaQueryWrapper<>();
        wrapper3.eq(Express::getStationId, id);
        wrapper3.eq(Express::getStatus, 2);
        long cancelledCount = expressService.count(wrapper3);
        
        statistics.put("stationName", station.getName());
        statistics.put("capacity", station.getCapacity());
        statistics.put("pendingCount", pendingCount);
        statistics.put("pickedCount", pickedCount);
        statistics.put("cancelledCount", cancelledCount);
        statistics.put("totalCount", pendingCount + pickedCount + cancelledCount);
        statistics.put("usageRate", String.format("%.2f", 
            (double)(pendingCount + pickedCount + cancelledCount) / station.getCapacity() * 100));
        
        return statistics;
    }
}
```

**关键业务逻辑**：
1. **创建站点流程**：
   - 检查站点名称唯一性
   - 设置默认容量和状态
   - 保存站点信息

2. **更新站点流程**：
   - 检查站点是否存在
   - 检查站点名称是否与其他站点重复
   - 更新站点信息

3. **删除站点流程**：
   - 检查站点是否存在
   - 检查站点下是否有未取件的快递
   - 如果有未取件快递，不允许删除
   - 删除站点

4. **站点统计流程**：
   - 统计待取件、已取件、已取消的快递数量
   - 计算总数量和使用率
   - 返回统计信息

```vue
<script setup>
import { ref, reactive, onMounted } from 'vue'
import { getStationList, createStation, updateStation, deleteStation, getStationStatistics } from '@/api/station'
import { ElMessage, ElMessageBox } from 'element-plus'

const loading = ref(false)
const dialogVisible = ref(false)
const statisticsDialogVisible = ref(false)
const dialogTitle = ref('创建站点')
const formRef = ref(null)

const searchForm = reactive({
  name: '',
  status: null
})

const pagination = reactive({
  page: 1,
  size: 10,
  total: 0
})

const tableData = ref([])
const stationForm = reactive({
  id: null,
  name: '',
  address: '',
  phone: '',
  capacity: 1000,
  status: 1
})

const statisticsData = reactive({
  stationName: '',
  capacity: 0,
  pendingCount: 0,
  pickedCount: 0,
  cancelledCount: 0,
  totalCount: 0,
  usageRate: '0'
})

const rules = {
  name: [
    { required: true, message: '请输入站点名称', trigger: 'blur' }
  ],
  address: [
    { required: true, message: '请输入站点地址', trigger: 'blur' }
  ],
  phone: [
    { required: true, message: '请输入联系电话', trigger: 'blur' },
    { pattern: /^1[3-9]\d{9}$/, message: '请输入正确的手机号', trigger: 'blur' }
  ],
  capacity: [
    { required: true, message: '请输入容量', trigger: 'blur' }
  ],
  status: [
    { required: true, message: '请选择状态', trigger: 'change' }
  ]
}

onMounted(() => {
  loadStationList()
})

async function loadStationList() {
  loading.value = true
  try {
    const res = await getStationList({
      page: pagination.page,
      size: pagination.size,
      ...searchForm
    })
    tableData.value = res.data.records
    pagination.total = res.data.total
  } catch (error) {
    ElMessage.error('加载站点列表失败')
  } finally {
    loading.value = false
  }
}

function handleSearch() {
  pagination.page = 1
  loadStationList()
}

function handleReset() {
  searchForm.name = ''
  searchForm.status = null
  handleSearch()
}

function handleCreate() {
  dialogTitle.value = '创建站点'
  stationForm.id = null
  stationForm.name = ''
  stationForm.address = ''
  stationForm.phone = ''
  stationForm.capacity = 1000
  stationForm.status = 1
  dialogVisible.value = true
}

function handleEdit(row) {
  dialogTitle.value = '编辑站点'
  stationForm.id = row.id
  stationForm.name = row.name
  stationForm.address = row.address
  stationForm.phone = row.phone
  stationForm.capacity = row.capacity
  stationForm.status = row.status
  dialogVisible.value = true
}

async function handleSubmit() {
  try {
    await formRef.value.validate()
    
    if (stationForm.id) {
      await updateStation(stationForm)
      ElMessage.success('更新成功')
    } else {
      await createStation(stationForm)
      ElMessage.success('创建成功')
    }
    
    dialogVisible.value = false
    loadStationList()
  } catch (error) {
    ElMessage.error(error.message || '操作失败')
  }
}

function handleView(row) {
  ElMessageBox.alert(
    `站点名称：${row.name}\n` +
    `站点地址：${row.address}\n` +
    `联系电话：${row.phone}\n` +
    `容量：${row.capacity} 件\n` +
    `状态：${row.status === 1 ? '启用' : '停用'}`,
    '站点详情',
    { confirmButtonText: '确定' }
  )
}

async function handleStatistics(row) {
  try {
    const res = await getStationStatistics(row.id)
    Object.assign(statisticsData, res.data)
    statisticsDialogVisible.value = true
  } catch (error) {
    ElMessage.error('获取统计信息失败')
  }
}

async function handleDelete(row) {
  try {
    await ElMessageBox.confirm('确定要删除该站点吗？', '提示', {
      confirmButtonText: '确定',
      cancelButtonText: '取消',
      type: 'warning'
    })
    await deleteStation(row.id)
    ElMessage.success('删除成功')
    loadStationList()
  } catch (error) {
    if (error !== 'cancel') {
      ElMessage.error(error.message || '删除失败')
    }
  }
```

**关键点**：

- 设置默认容量和状态
- 表单验证

### 2. 删除站点流程

```
管理员点击"删除"按钮
    ↓
确认删除操作
    ↓
调用删除API（DELETE /api/station/delete/{id}）
    ↓
后端检查站点是否存在
    ↓
检查站点下是否有未取件的快递
    ↓

    ↓
```

- 检查站点下是否有未取件快递

## 📝 学习要点总结

### 必须掌握的概念

1. **站点管理**：站点的增删改查操作
2. **数据关联**：站点与快递、快递员的关联关系
3. **业务规则**：站点下有未取件快递时不能删除
4. **统计功能**：站点快递数量统计和使用率计算
5. **状态管理**：站点的启用和停用状态

### 答辩要点

1. **为什么站点下有未取件快递时不能删除？**
   - 保证数据完整性
   - 防止快递丢失
   - 避免业务中断

2. **如何计算站点的使用率？**
   - 统计站点下所有快递数量
   - 除以站点容量
   - 乘以100得到百分比

3. **站点与快递的关系是什么？**
   - 一对多关系
   - 一个站点可以有多个快递
   - 每个快递必须属于一个站点

4. **如何优化站点管理？**
   - 增加站点地图展示
   - 增加站点容量预警
   - 增加站点运营数据分析

## ❓ 常见问题

**Q1：站点容量满了怎么办？**
A：可以增加站点容量，或者创建新的站点。

**Q2：如何查看站点的详细运营数据？**

A：不能，停用的站点不能接收新的快递，但已有的快递可以正常取件。

A：点击"编辑"按钮，修改站点信息后保存。
