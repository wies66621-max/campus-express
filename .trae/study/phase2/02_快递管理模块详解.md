```java
package com.school.campusexpress.controller;

import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.school.campusexpress.common.Result;
import com.school.campusexpress.dto.ExpressDTO;
import com.school.campusexpress.entity.Express;
import com.school.campusexpress.service.ExpressService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/api/express")
@CrossOrigin
public class ExpressController {
    
    @Autowired
    private ExpressService expressService;
    
    /**
     * 创建快递（入库）
     */
    @PostMapping("/create")
    public Result<?> createExpress(@RequestBody ExpressDTO expressDTO) {
        try {
            Express express = expressService.createExpress(expressDTO);
            return Result.success(express);
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 根据取件码取件
     */
    @PostMapping("/pickup")
    public Result<?> pickupByCode(@RequestBody PickupDTO pickupDTO) {
        try {
            Express express = expressService.pickupByCode(
                pickupDTO.getPickupCode(), 
                pickupDTO.getUserId()
            );
            return Result.success(express);
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 获取快递列表
     */
    @GetMapping("/list")
    public Result<?> getExpressList(
        @RequestParam(defaultValue = "1") Integer page,
        @RequestParam(defaultValue = "10") Integer size,
        @RequestParam(required = false) String expressNumber,
        @RequestParam(required = false) String recipientName,
        @RequestParam(required = false) Integer status,
        @RequestParam(required = false) Long stationId
    ) {
        try {
            Page<Express> result = expressService.getExpressList(
                page, size, expressNumber, recipientName, status, stationId
            );
            return Result.success(result);
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 获取快递详情
     */
    @GetMapping("/detail/{id}")
    public Result<?> getExpressDetail(@PathVariable Long id) {
        try {
            Express express = expressService.getById(id);
            return Result.success(express);
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 更新快递信息
     */
    @PutMapping("/update")
    public Result<?> updateExpress(@RequestBody Express express) {
        try {
            expressService.updateById(express);
            return Result.success("更新成功");
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
    
    /**
     * 取消快递
     */
    @PutMapping("/cancel/{id}")
    public Result<?> cancelExpress(@PathVariable Long id) {
        try {
            expressService.cancelExpress(id);
            return Result.success("取消成功");
        } catch (Exception e) {
            return Result.error(e.getMessage());
        }
    }
}
```

**关键接口说明**：
- `POST /api/express/create`：创建快递（入库）
- `POST /api/express/pickup`：根据取件码取件
- `GET /api/express/list`：获取快递列表（支持分页和搜索）
- `GET /api/express/detail/{id}`：获取快递详情
- `PUT /api/express/update`：更新快递信息
- `PUT /api/express/cancel/{id}`：取消快递

### 3. 快递服务实现（ExpressServiceImpl.java）

**文件位置**：`src/main/java/com/school/campusexpress/service/impl/ExpressServiceImpl.java`

```java
package com.school.campusexpress.service.impl;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.baomidou.mybatisplus.extension.service.impl.ServiceImpl;
import com.school.campusexpress.dto.ExpressDTO;
import com.school.campusexpress.entity.Express;
import com.school.campusexpress.entity.PickupRecord;
import com.school.campusexpress.mapper.ExpressMapper;
import com.school.campusexpress.service.ExpressService;
import com.school.campusexpress.service.PickupRecordService;
import com.school.campusexpress.util.PickupCodeUtil;
import org.springframework.beans.BeanUtils;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import org.springframework.util.StringUtils;

@Service
public class ExpressServiceImpl extends ServiceImpl<ExpressMapper, Express> implements ExpressService {
    
    @Autowired
    private PickupRecordService pickupRecordService;
    
    @Override
    @Transactional
    public Express createExpress(ExpressDTO expressDTO) {
        // 检查快递单号是否已存在
        LambdaQueryWrapper<Express> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(Express::getExpressNumber, expressDTO.getExpressNumber());
        Express existExpress = getOne(wrapper);
        
        if (existExpress != null) {
            throw new RuntimeException("快递单号已存在");
        }
        
        // 创建快递
        Express express = new Express();
        BeanUtils.copyProperties(expressDTO, express);
        
        // 生成取件码
        String pickupCode = PickupCodeUtil.generate();
        express.setPickupCode(pickupCode);
        
        // 设置初始状态
        express.setStatus(0); // 待取件
        
        save(express);
        
        return express;
    }
    
    @Override
    @Transactional
    public Express pickupByCode(String pickupCode, Long userId) {
        // 验证取件码
        if (pickupCode == null || pickupCode.trim().isEmpty()) {
            throw new RuntimeException("取件码不能为空");
        }
        
        // 查询快递
        LambdaQueryWrapper<Express> wrapper = new LambdaQueryWrapper<>();
        wrapper.eq(Express::getPickupCode, pickupCode);
        Express express = getOne(wrapper);
        
        if (express == null) {
            throw new RuntimeException("取件码不存在");
        }
        
        // 检查快递状态
        if (express.getStatus() != 0) {
            throw new RuntimeException("快递已被取件或已取消");
        }
        
        // 更新快递状态
        express.setStatus(1); // 已取件
        express.setUserId(userId);
        updateById(express);
        
        // 创建取件记录
        PickupRecord record = new PickupRecord();
        record.setExpressId(express.getId());
        record.setOperatorId(userId);
        record.setPickupTime(LocalDateTime.now());
        record.setStatus(1);
        record.setRemark("取件成功");
        record.setCreateTime(LocalDateTime.now());
        pickupRecordService.save(record);
        
        return express;
    }
    
    @Override
    public Page<Express> getExpressList(Integer page, Integer size, String expressNumber, 
                                         String recipientName, Integer status, Long stationId) {
        Page<Express> pageParam = new Page<>(page, size);
        LambdaQueryWrapper<Express> wrapper = new LambdaQueryWrapper<>();
        
        if (StringUtils.hasText(expressNumber)) {
            wrapper.like(Express::getExpressNumber, expressNumber);
        }
        
        if (StringUtils.hasText(recipientName)) {
            wrapper.like(Express::getRecipientName, recipientName);
        }
        
        if (status != null) {
            wrapper.eq(Express::getStatus, status);
        }
        
        if (stationId != null) {
            wrapper.eq(Express::getStationId, stationId);
        }
        
        wrapper.orderByDesc(Express::getCreateTime);
        
        return page(pageParam, wrapper);
    }
    
    @Override
    @Transactional
    public void cancelExpress(Long id) {
        Express express = getById(id);
        
        if (express == null) {
            throw new RuntimeException("快递不存在");
        }
        
        if (express.getStatus() != 0) {
            throw new RuntimeException("只能取消待取件的快递");
        }
        
        express.setStatus(2); // 已取消
        updateById(express);
    }
}
```

**关键业务逻辑**：
1. **创建快递流程**：
   - 检查快递单号唯一性
   - 生成6位数字取件码
   - 设置初始状态为待取件

2. **取件流程**：
   - 验证取件码
   - 检查快递状态
   - 更新快递状态为已取件
   - 创建取件记录
   - 使用事务保证数据一致性

3. **查询流程**：
   - 支持多条件组合查询
   - 支持分页
   - 按创建时间倒序排列

### 4. 取件码工具类（PickupCodeUtil.java）

**文件位置**：`src/main/java/com/school/campusexpress/util/PickupCodeUtil.java`

```java
package com.school.campusexpress.util;

import java.util.Random;

public class PickupCodeUtil {
    
    private static final Random random = new Random();
    
    /**
     * 生成6位数字取件码
     */
    public static String generate() {
        return String.format("%06d", random.nextInt(1000000));
    }
    
    /**
     * 验证取件码格式
     */
    public static boolean isValid(String pickupCode) {
        if (pickupCode == null || pickupCode.length() != 6) {
            return false;
        }
        
        return pickupCode.matches("\\d{6}");
    }
}
```

**取件码机制说明**：
- 生成6位纯数字取件码
- 取件码范围：000000-999999
- 总共100万个可能的取件码
- 简单易记，便于用户使用

## 📝 学习要点总结

### 必须掌握的概念

1. **取件码机制**：6位数字，简单易记，便于用户使用
2. **事务管理**：取件操作使用事务保证数据一致性
3. **状态管理**：快递有三种状态（待取件、已取件、已取消）
4. **分页查询**：使用MyBatis-Plus的分页插件
5. **多条件搜索**：支持按快递单号、收件人、状态等条件组合查询

### 答辩要点

1. **取件码如何生成？**
   - 使用Random类生成6位随机数字
   - 格式化为6位字符串（不足补0）
   - 范围：000000-999999

2. **如何保证取件操作的原子性？**
   - 使用Spring的@Transactional注解
   - 更新快递状态和创建取件记录在同一事务中
   - 任何一步失败都会回滚

3. **如何防止重复取件？**
   - 检查快递状态，只有待取件状态才能取件
   - 取件后立即更新状态为已取件
   - 数据库层面添加状态检查

4. **取件码会重复吗？**
   - 理论上可能重复，但概率极低（1/1000000）
   - 实际使用中可以通过检查数据库避免重复
   - 可以在生成时检查是否已存在

## ❓ 常见问题

**Q1：取件码忘记了怎么办？**
A：可以通过快递单号或收件人信息查询快递，查看取件码。

**Q2：如何防止他人冒领取件？**
A：当前版本仅通过取件码验证，可以增加手机号验证、身份证验证等安全措施。

**Q3：快递可以取消吗？**
A：只有待取件状态的快递可以取消，已取件的快递不能取消。

**Q4：取件码的有效期是多久？**
A：当前版本没有设置有效期，可以在快递入库后随时取件。
